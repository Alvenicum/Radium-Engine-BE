12 WeightMatrix computeSTBS_weights( const Vector3Array& inMesh, const Skeleton& skel ) {
13  std::vector<Eigen::Triplet<Scalar>> triplets;
14  for ( int i = 0; i < int( inMesh.size() ); ++i ) {
15  const auto& pi = inMesh[uint( i )];
16  for ( int j = 0; j < int( skel.size() ); ++j ) {
17  Vector3 a, b;
18  skel.getBonePoints( uint( j ), a, b );
19  const Vector3 ab = b - a;
20  Scalar t = Geometry::projectOnSegment( pi, a, ab );
21  if ( t > 0 ) { triplets.push_back( Eigen::Triplet<Scalar>( i, j, t ) ); }
22  }
23  }
24  WeightMatrix weights( uint( inMesh.size() ), skel.size() );
25  weights.setFromTriplets( triplets.begin(), triplets.end() );
26  return weights;
27 }

29 void linearBlendSkinningSTBS( const SkinningRefData& refData,
30  const Vector3Array& tangents,
31  const Vector3Array& bitangents,
32  SkinningFrameData& frameData ) {
33  // reset all outputs
34 #pragma omp parallel for
35  for ( int i = 0; i < int( frameData.m_currentPosition.size() ); ++i ) {
36  frameData.m_currentPosition[i] = Vector3::Zero();
37  frameData.m_currentNormal[i] = Vector3::Zero();
38  frameData.m_currentTangent[i] = Vector3::Zero();
39  frameData.m_currentBitangent[i] = Vector3::Zero();
40  }
41  // get all needed data
42  const auto& refPose = refData.m_skeleton.getPose( HandleArray::SpaceType::MODEL );
43  const auto& curPose = frameData.m_skeleton.getPose( HandleArray::SpaceType::MODEL );
44  const auto relPose = relativePose( curPose, refPose );
45  const auto& vertices = refData.m_referenceMesh.vertices();
46  const auto& normals = refData.m_referenceMesh.normals();
47  // for each bone
48  for ( uint k = 0; k < refData.m_weights.outerSize(); ++k ) {
49  // get the bone's full stretch vector
50  Vector3 a, b, a_, b_;
51  refData.m_skeleton.getBonePoints( k, a, b );
52  frameData.m_skeleton.getBonePoints( k, a_, b_ );
53  const Vector3 ab = b - a;
54  const Vector3 es = ( std::sqrt( ( b_ - a_ ).squaredNorm() / ab.squaredNorm() ) - 1 ) * ab;
55  // fetch bone transforms
56  const auto R_rel = relPose[k].linear();
57  const auto& T_ref = refPose[k];
58  const auto& T_cur = curPose[k];
59  const auto& B = refData.m_bindMatrices[k];
60  // then go through the skinned vertices
61  const int nonZero = refData.m_weights.col( k ).nonZeros();
62  WeightMatrix::InnerIterator it0( refData.m_weights, k );
63 #pragma omp parallel for
64  for ( int nz = 0; nz < nonZero; ++nz ) {
65  WeightMatrix::InnerIterator it = it0 + Eigen::Index( nz );
66  const uint i = it.row();
67  const uint j = it.col();
68  const Scalar w = it.value();
69  const Scalar eis = refData.m_weightSTBS.coeff( i, j );
70  frameData.m_currentPosition[i] +=
71  w * ( refData.m_meshTransformInverse *
72  ( a_ + R_rel * ( eis * es - a + T_ref * B * vertices[i] ) ) );
73  auto J = refData.m_meshTransformInverse * T_cur * B;
74  frameData.m_currentNormal[i] += w * J.linear() * normals[i];
75  frameData.m_currentTangent[i] += w * J.linear() * tangents[i];
76  frameData.m_currentBitangent[i] += w * J.linear() * bitangents[i];
77  }
78  }

81 DQList computeDQSTBS( const SkinningRefData& refData, const Skeleton& poseSkel ) {
82  const auto& refPose = refData.m_skeleton.getPose( HandleArray::SpaceType::MODEL );
83  const auto& curPose = poseSkel.getPose( HandleArray::SpaceType::MODEL );
84  const auto relPose = relativePose( curPose, refPose );
85  const auto& weight = refData.m_weights;
86  const auto& weightSTBS = refData.m_weightSTBS;
87  CORE_ASSERT( ( refPose.size() == size_t( weight.cols() ) ), "pose/weight size mismatch." );
88  DQList DQ( uint( weight.rows() ),
89  DualQuaternion( Quaternion( 0, 0, 0, 0 ), Quaternion( 0, 0, 0, 0 ) ) );
90  // Stores the first non-zero quaternion for each vertex.
91  std::vector<uint> firstNonZero( weight.rows(), std::numeric_limits<uint>::max() );
92  // Contains the converted dual quaternions from the pose
93  std::vector<DualQuaternion> poseDQ( relPose.size() );
94 
95  // for each bone
96  for ( uint k = 0; k < weight.outerSize(); ++k ) {
97  // get the bone's full stretch vector "e_s"
98  Vector3 a, b, a_, b_;
99  refData.m_skeleton.getBonePoints( k, a, b );
100  poseSkel.getBonePoints( k, a_, b_ );
101  const Vector3 ab = b - a;
102  const Vector3 es = ( std::sqrt( ( b_ - a_ ).squaredNorm() / ab.squaredNorm() ) - 1 ) * ab;
103  // fetch bone transforms
104  const auto R_rel = relPose[k].linear();
105  const auto& T_ref = refPose[k];
106  const auto& B = refData.m_bindMatrices[k];
107  poseDQ[k] = DualQuaternion( refData.m_meshTransformInverse * curPose[k] * B );
108  // then go through the skinned vertices
109  const int nonZero = weight.col( k ).nonZeros();
110  WeightMatrix::InnerIterator it0( weight, k );
111 #pragma omp parallel for
112  for ( int nz = 0; nz < nonZero; ++nz ) {
113  WeightMatrix::InnerIterator itn = it0 + Eigen::Index( nz );
114  const uint i = itn.row();
115  const uint j = itn.col();
116  const Scalar w = itn.value();
117  const Scalar eis = weightSTBS.coeff( i, j );
118  // build the dual quaternion from the STBS transform
119  Transform Ti = Transform::Identity();
120  Ti.translation() = a_ + R_rel * ( eis * es - a );
121  const Transform Ri( R_rel * T_ref * B );
122  DualQuaternion D( refData.m_meshTransformInverse * Ti * Ri );
123  D.normalize();
124  // mix it with the dual quaternion from the other skinning bones
125  firstNonZero[i] = std::min( firstNonZero[i], uint( k ) );
126  const Scalar sign =
127  Math::signNZ( poseDQ[j].getQ0().dot( poseDQ[firstNonZero[i]].getQ0() ) );
128  DQ[i] += D * w * sign;
129  }
130  }
131  // Normalize all dual quats.
132 #pragma omp parallel for
133  for ( int i = 0; i < int( DQ.size() ); ++i ) {
134  DQ[i].normalize();
135  }
136  return DQ;
137 }

139 void RA_CORE_API dualQuaternionSkinningSTBS( const SkinningRefData& refData,
140  const Vector3Array& tangents,
141  const Vector3Array& bitangents,
142  SkinningFrameData& frameData ) {
143  // compute the STBS dual quaternion for each vertex
144  const auto DQ = computeDQSTBS( refData, frameData.m_skeleton );
145  // apply DQS
146  const auto& vertices = refData.m_referenceMesh.vertices();
147  const auto& normals = refData.m_referenceMesh.normals();
148 #pragma omp parallel for
149  for ( int i = 0; i < int( frameData.m_currentPosition.size() ); ++i ) {
150  const auto& DQi = DQ[i];
151  frameData.m_currentPosition[i] = DQi.transform( vertices[i] );
152  frameData.m_currentNormal[i] = DQi.rotate( normals[i] );
153  frameData.m_currentTangent[i] = DQi.rotate( tangents[i] );
154  frameData.m_currentBitangent[i] = DQi.rotate( bitangents[i] );
