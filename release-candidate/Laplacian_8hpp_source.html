37// Defining the LaplacianMatrix as the sparse matrix such that:
38// L = f( D, A )
39// where f( D, A ) is a function taking in input a DegreeMatrix and an AdjacencyMatrix
40using LaplacianMatrix = Sparse;

42/*
43 * Return the LaplacianMatrix from the given matrices D and A.
44 *
45 * If POSITIVE_SEMI_DEFINITE is true then:
46 * L = D - A
47 * where L could/should be a positive semi-definite matrix.
48 *
49 * If POSITIVE_SEMI_DEFINITE is false then:
50 * L = A - D
51 * where L could/should be a negative semi-definite matrix.
52 */
55LaplacianMatrix RA_CORE_API standardLaplacian( const DegreeMatrix& D,
56 const AdjacencyMatrix& A,
57 const bool POSITIVE_SEMI_DEFINITE = true );
58
59/*
60 * Return the LaplacianMatrix from the given matrices D and A.
61 *
62 * The LaplacianMatrix is defined as:
63 * L = I - D^-1/2 A D^-1/2
64 * where I is the identity matrix.
65 */
66LaplacianMatrix RA_CORE_API symmetricNormalizedLaplacian( const DegreeMatrix& D,
67 const AdjacencyMatrix& A );
68
69/*
70 * Return the LaplacianMatrix from the given matrices D and A.
71 *
72 * The LaplacianMatrix is defined as:
73 * L = I - D^-1 A
74 * where I is the identity matrix.
75 */
76LaplacianMatrix RA_CORE_API randomWalkNormalizedLaplacian( const DegreeMatrix& D,
77 const AdjacencyMatrix& A );

79/*
80 * Return the LaplacianMatrix computed as the k-power of the matrix L.

82 * The definition was taken from:

88LaplacianMatrix RA_CORE_API powerLaplacian( const LaplacianMatrix& L, const uint k );

90/*
91 * Return the LaplacianMatrix vector for the given set of points and triangles.
92 *
93 * The LaplacianMatrix is defined as:
94 * L( i, j ) = -0.5 * sum( cot( alpha_ij ) + cot( beta_ij ) ) , with i != j and exist edge {
95 * i, j } L( i, i ) = 0.5 * sum( L( i, j ) ) , with i != j where alpha_ij
96 * and beta_ij are the angles opposite the edge { i, j }, and L could/should be a positive
97 * semi-definite matrix.
98 */
100LaplacianMatrix RA_CORE_API cotangentWeightLaplacian( const VectorArray<Vector3>& p,
101 const AlignedStdVector<Vector3ui>& T );
