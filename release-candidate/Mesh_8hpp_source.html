445 template <typename CoreMeshType>
446 CoreMeshType createCoreMeshFromGeometryData( const Ra::Core::Asset::GeometryData* data ) {
447  CoreMeshType mesh;
448  typename CoreMeshType::PointAttribHandle::Container vertices;
449  typename CoreMeshType::NormalAttribHandle::Container normals;
450  typename CoreMeshType::IndexContainerType indices;
451 
452  vertices.reserve( data->getVerticesSize() );
453  std::copy(
454  data->getVertices().begin(), data->getVertices().end(), std::back_inserter( vertices ) );
455 
456  if ( data->hasNormals() ) {
457  normals.reserve( data->getVerticesSize() );
458  std::copy(
459  data->getNormals().begin(), data->getNormals().end(), std::back_inserter( normals ) );
460  }
461 
462  const auto& faces = data->getFaces();
463  indices.reserve( faces.size() );
464  std::copy( faces.begin(), faces.end(), std::back_inserter( indices ) );
465  mesh.setVertices( std::move( vertices ) );
466  mesh.setNormals( std::move( normals ) );
467 
468  // \todo remove when data will handle all the attributes in a coherent way.
469  if ( data->hasTangents() ) {
470  mesh.addAttrib( Ra::Core::Geometry::getAttribName( Ra::Core::Geometry::VERTEX_TANGENT ),
471  data->getTangents() );
472  }
473 
474  if ( data->hasBiTangents() ) {
475  mesh.addAttrib( Ra::Core::Geometry::getAttribName( Ra::Core::Geometry::VERTEX_BITANGENT ),
476  data->getBiTangents() );
477  }
478 
479  if ( data->hasTextureCoordinates() ) {
480  mesh.addAttrib( Ra::Core::Geometry::getAttribName( Ra::Core::Geometry::VERTEX_TEXCOORD ),
481  data->getTexCoords() );
482  }
483 
484  // add custom attribs
485  // only attributs not handled before are handled by data->getAttribManager()
486  // but futur plan will handle also "usual" attibutes this way
487  mesh.vertexAttribs().copyAllAttributes( data->getAttribManager() );
488 
489  // also this one will be automatically done with the futur behavior
490  // mesh->addData( Ra::Core::Geometry::VERTEX_WEIGHTS, meshData.weights );
491 
492  mesh.setIndices( std::move( indices ) );
493 
494  return mesh;
495 }
