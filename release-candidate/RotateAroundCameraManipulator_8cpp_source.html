33void rotateAroundPoint( Ra::Core::Asset::Camera* cam,
34 Ra::Core::Vector3& target,
35 Ra::Core::Quaternion& rotation,
36 const Ra::Core::Vector3& point ) {
37 Ra::Core::Vector3 t = cam->getPosition();
38 Scalar l = ( target - t ).norm();
39 Ra::Core::Transform inverseCamRotateAround;
40 Ra::Core::AngleAxis aa0 { rotation };
41 Ra::Core::AngleAxis aa1 { aa0.angle(), ( cam->getFrame().linear() * aa0.axis() ).normalized() };
42 inverseCamRotateAround.setIdentity();
43 inverseCamRotateAround.rotate( aa1 );
44 cam->applyTransform( inverseCamRotateAround );
45
46 Ra::Core::Vector3 trans = point + inverseCamRotateAround * ( t - point );
47 cam->setPosition( trans );
48 target = cam->getPosition() + l * cam->getDirection();
49}

55bool RotateAroundCameraManipulator::handleMouseMoveEvent(
56 QMouseEvent* event,
57 const Qt::MouseButtons& /*buttons*/,

60
61 Scalar dx = ( event->pos().x() - m_lastMouseX ) / m_camera->getWidth();
62 Scalar dy = ( event->pos().y() - m_lastMouseY ) / m_camera->getHeight();
63
64 if ( m_currentAction == TRACKBALLCAMERA_ROTATE )
65 handleCameraRotate( event->pos().x(), event->pos().y() );

68 else if ( m_currentAction == TRACKBALLCAMERA_ZOOM )
69 handleCameraZoom( dx, dy );
70 else if ( m_currentAction == TRACKBALLCAMERA_MOVE_FORWARD )
71 handleCameraMoveForward( dx, dy );

73 m_lastMouseX = event->pos().x();
74 m_lastMouseY = event->pos().y();
75
76 if ( m_light != nullptr ) {
77 m_light->setPosition( m_camera->getPosition() );
78 m_light->setDirection( m_camera->getDirection() );

111void RotateAroundCameraManipulator::alignOnClosestAxis() {
112 Ra::Core::Vector3 x( 1_ra, 0_ra, 0_ra );
113 Ra::Core::Vector3 y( 0_ra, 1_ra, 0_ra );
114 Ra::Core::Vector3 z( 0_ra, 0_ra, 1_ra );
115 Ra::Core::Vector3 oldDirection, newDirection = m_camera->getDirection();
116 Ra::Core::Vector3 newUpVector = m_camera->getUpVector();
117 // getFrame().inverse() is a transform we can apply on Vector3
118 Ra::Core::Vector3 pivotInCamSpace = m_camera->getFrame().inverse() * m_pivot;
119
120 auto updateMaxAndAxis =
121 []( Ra::Core::Vector3 ref, Ra::Core::Vector3 axis, Ra::Core::Vector3& out, Scalar& max ) {
122 Scalar d = ref.dot( axis );
123 if ( d > max ) {
124 max = d;
125 out = axis;
126 }
127 };
128
129 Scalar max = 0_ra;
130 Ra::Core::Vector3 ref = m_camera->getDirection();
131 for ( auto axis : std::vector<Ra::Core::Vector3> { -x, x, -y, y, -z, z } ) {
132 updateMaxAndAxis( ref, axis, newDirection, max );
133 }
134 m_camera->setDirection( newDirection );
135
136 max = 0_ra;
137 ref = m_camera->getUpVector();
138 for ( auto axis : std::vector<Ra::Core::Vector3> { -x, x, -y, y, -z, z } ) {
139 updateMaxAndAxis( ref, axis, newUpVector, max );
140 }
141 m_camera->setUpVector( newUpVector );
142
143 Ra::Core::Vector3 newPivot = m_camera->getFrame() * pivotInCamSpace;
144 Ra::Core::Vector3 trans = m_pivot - newPivot;
145 m_camera->setPosition( m_camera->getPosition() + trans );
146
147 if ( m_light != nullptr ) {
148 m_light->setPosition( m_camera->getPosition() );
149 m_light->setDirection( m_camera->getDirection() );
150 }
151}

169Scalar RotateAroundCameraManipulator::projectOnBall( Scalar x, Scalar y ) {
170 const Scalar size = 1.0;
171 const Scalar size2 = size * size;
172 const Scalar size_limit = size2 * 0.5;
173
174 const Scalar d = x * x + y * y;
175 return d < size_limit ? std::sqrt( size2 - d ) : size_limit / std::sqrt( d );
176}

178Ra::Core::Quaternion
179RotateAroundCameraManipulator::deformedBallQuaternion( Scalar x, Scalar y, Scalar cx, Scalar cy ) {
180 // Points on the deformed ball
181 Scalar px = m_cameraSensitivity * ( m_lastMouseX - cx ) / m_camera->getWidth();
182 Scalar py = m_cameraSensitivity * ( cy - m_lastMouseY ) / m_camera->getHeight();
183 Scalar dx = m_cameraSensitivity * ( x - cx ) / m_camera->getWidth();
184 Scalar dy = m_cameraSensitivity * ( cy - y ) / m_camera->getHeight();
185
186 const Ra::Core::Vector3 p1( px, py, projectOnBall( px, py ) );
187 const Ra::Core::Vector3 p2( dx, dy, projectOnBall( dx, dy ) );
188 // Approximation of rotation angle
189 // Should be divided by the projectOnBall size, but it is 1.0
190 Ra::Core::Vector3 axis = p2.cross( p1 );
191
192 if ( axis.norm() > 10_ra * Ra::Core::Math::machineEps ) {
193 const Scalar angle =
194 5.0_ra *
195 std::asin( std::sqrt( axis.squaredNorm() / p1.squaredNorm() / p2.squaredNorm() ) );
196 return Ra::Core::Quaternion( Ra::Core::AngleAxis( angle, axis.normalized() ) );
197 }
198 return Ra::Core::Quaternion {
199 Ra::Core::AngleAxis( 0_ra, Ra::Core::Vector3( 0_ra, 0_ra, 1_ra ) ) };
200}

210void RotateAroundCameraManipulator::handleCameraPan( Scalar dx, Scalar dy ) {
211 Scalar x = dx * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;
212 Scalar y = dy * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;
213 // Move camera and trackball center, keep the distance to the center
214 Ra::Core::Vector3 R = -m_camera->getRightVector();
215 Ra::Core::Vector3 U = m_camera->getUpVector();
216
217 Ra::Core::Transform T( Ra::Core::Transform::Identity() );
218 Ra::Core::Vector3 t = x * R + y * U;
219 T.translate( t );
220 m_target += t;
221
222 m_camera->applyTransform( T );
223}
