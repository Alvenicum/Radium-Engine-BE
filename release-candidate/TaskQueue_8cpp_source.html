44void TaskQueue::addDependency( TaskQueue::TaskId predecessor, TaskQueue::TaskId successor ) {
45 CORE_ASSERT( predecessor.isValid() && ( predecessor < m_tasks.size() ),
46 "Invalid predecessor task" );
47 CORE_ASSERT( successor.isValid() && ( successor < m_tasks.size() ), "Invalid successor task" );
48 CORE_ASSERT( predecessor != successor, "Cannot add self-dependency" );
49
50 CORE_ASSERT( std::find( m_dependencies[predecessor].begin(),
51 m_dependencies[predecessor].end(),
52 successor ) == m_dependencies[predecessor].end(),
53 "Cannot add a dependency twice" );
54
55 m_dependencies[predecessor].push_back( successor );
56 ++m_remainingDependencies[successor];

90void TaskQueue::resolveDependencies() {
91 for ( const auto& pre : m_pendingDepsPre ) {
92 ON_ASSERT( bool result = ) addDependency( pre.first, pre.second );
93 CORE_WARN_IF( !result,

95 << pre.second << ")" );
96 }
97 for ( const auto& pre : m_pendingDepsSucc ) {

113void TaskQueue::detectCycles() {
114#if defined( CORE_DEBUG )
115 // Do a depth-first search of the nodes.
116 std::vector<bool> visited( m_tasks.size(), false );
117 std::stack<TaskId> pending;
118
119 for ( uint id = 0; id < m_tasks.size(); ++id ) {
120 if ( m_dependencies[id].size() == 0 ) { pending.push( TaskId( id ) ); }
121 }
122
123 // If you hit this assert, there are tasks in the list but
124 // all tasks have dependencies so no task can start.
125 CORE_ASSERT( m_tasks.empty() || !pending.empty(), "No free tasks." );
126
127 while ( !pending.empty() ) {
128 TaskId id = pending.top();
129 pending.pop();
130
131 // The task has already been visited. It means there is a cycle in the task graph.
132 CORE_ASSERT( !( visited[id] ), "Cycle detected in tasks !" );
133
134 visited[id] = true;
135 for ( const auto& dep : m_dependencies[id] ) {
136 pending.push( dep );
137 }
138 }
139#endif
140}

142void TaskQueue::startTasks() {
143 // Add pending dependencies.

154 // Wake up all threads.

182void TaskQueue::runThread( uint id ) {
183 while ( true ) {
184 TaskId task;
185
186 // Acquire mutex.
187 {
188 std::unique_lock<std::mutex> lock( m_taskQueueMutex );
189
190 // Wait for a new task
191 // TODO : use the second form of wait()
192 while ( !m_shuttingDown && m_taskQueue.empty() ) {
193 m_threadNotifier.wait( lock );
194 }
195 // If the task queue is shutting down we quit, releasing
196 // the lock.
197 if ( m_shuttingDown ) { return; }
198
199 // If we are here it means we got a task
200 task = m_taskQueue.back();
201 m_taskQueue.pop_back();
202 ++m_processingTasks;
203 CORE_ASSERT( task.isValid() && task < m_tasks.size(), "Invalid task" );
204 }
205 // Release mutex.
206
207 // Run task
208 m_timerData[task].start = Utils::Clock::now();
209 m_timerData[task].threadId = id;
210 m_tasks[task]->process();
211 m_timerData[task].end = Utils::Clock::now();
212
213 // Critical section : mark task as finished and en-queue dependencies.
214 uint newTasks = 0;
215 {
216 std::unique_lock<std::mutex> lock( m_taskQueueMutex );
217 for ( auto t : m_dependencies[task] ) {
218 uint& nDepends = m_remainingDependencies[t];
219 CORE_ASSERT( nDepends > 0, "Inconsistency in dependencies" );
220 --nDepends;
221 if ( nDepends == 0 ) {
222 queueTask( t );
223 ++newTasks;
224 }
225 // TODO :Easy optimization : grab one of the new task and process it immediately.
226 }
227 --m_processingTasks;
228 }
229 // If we added new tasks, we wake up one thread to execute it.
230 if ( newTasks > 0 ) { m_threadNotifier.notify_one(); }
231 } // End of while(true)
232}
