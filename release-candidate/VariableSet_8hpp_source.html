468 template <typename T>
469 void VariableSet::addVariableType() {
470  assert( !existsVariableType<T>() );
471  // used to remove all stored data at deletion time
472  m_clearFunctions.emplace_back( []( VariableSet& c ) { m_variables<T>.erase( &c ); } );
473  // used to copy the stored data when copying the object
474  m_copyFunctions.emplace_back( []( const VariableSet& from, VariableSet& to ) {
475  m_variables<T>[&to] = m_variables<T>[&from];
476  } );
477  m_moveFunctions.emplace_back( []( VariableSet& from, VariableSet& to ) {
478  m_variables<T>[&to] = std::move( m_variables<T>[&from] );
479  } );
480  // used to merge (keep) the stored data from container "from" to container "to"
481  m_mergeKeepFunctions.emplace_back( []( const VariableSet& from, VariableSet& to ) {
482  if ( !to.existsVariableType<T>() ) { to.addVariableType<T>(); }
483  for ( const auto& t : m_variables<T>[&from] ) {
484  m_variables<T>[&to].insert( t );
485  }
486  } );
487  // used to merge (replace) the stored data from container "from" to container "to"
488  m_mergeReplaceFunctions.emplace_back( []( const VariableSet& from, VariableSet& to ) {
489  if ( !to.existsVariableType<T>() ) { to.addVariableType<T>(); }
490  for ( const auto& t : m_variables<T>[&from] ) {
491  m_variables<T>[&to].insert_or_assign( t.first, t.second );
492  }
493  } );
494  // use to compute gauge on the stored data
495  m_sizeFunctions.emplace_back(
496  []( const VariableSet& c ) { return m_variables<T>[&c].size(); } );
497  // used to call dynamically typed visitors
498  m_visitFunctions.emplace_back( []( const VariableSet& c, DynamicVisitor& v ) {
499  for ( auto&& t : m_variables<T>[&c] ) {
500  v( { std::ref( t ) } );
501  }
502  } );
503  // remember the stored type and its rank
504  m_storedType.emplace_back( std::type_index( typeid( T ) ) );
505 }

513 template <typename T>
514 bool VariableSet::deleteAllVariables() {
515  auto iter = m_variables<T>.find( this );
516  if ( iter != m_variables<T>.end() ) {
517  auto tidx = std::type_index( typeid( T ) );
518  auto it = std::find( m_storedType.begin(), m_storedType.end(), tidx );
519  auto idx = it - m_storedType.begin();
520  m_clearFunctions.erase( m_clearFunctions.begin() + idx );
521  m_copyFunctions.erase( m_copyFunctions.begin() + idx );
522  m_moveFunctions.erase( m_moveFunctions.begin() + idx );
523  m_mergeKeepFunctions.erase( m_mergeKeepFunctions.begin() + idx );
524  m_mergeReplaceFunctions.erase( m_mergeReplaceFunctions.begin() + idx );
525  m_sizeFunctions.erase( m_sizeFunctions.begin() + idx );
526  m_visitFunctions.erase( m_visitFunctions.begin() + idx );
527  m_storedType.erase( it );
528  m_variables<T>.erase( iter );
529  return true;
530  }
531  return false;
532 }
