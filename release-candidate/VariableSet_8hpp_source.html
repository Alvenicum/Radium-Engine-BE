580 template <typename T>
581 void VariableSet::addVariableType() {
582  assert( !existsVariableType<T>() );
583  // used to remove all stored data at deletion time
584  m_clearFunctions.emplace_back( []( VariableSet& c ) { m_variables<T>.erase( &c ); } );
585  // used to copy the stored data when copying the object
586  m_copyFunctions.emplace_back( []( const VariableSet& from, VariableSet& to ) {
587  m_variables<T>[&to] = m_variables<T>[&from];
588  } );
589  m_moveFunctions.emplace_back( []( VariableSet& from, VariableSet& to ) {
590  m_variables<T>[&to] = std::move( m_variables<T>[&from] );
591  m_variables<T>.erase( &from );
592  } );
593  // used to merge (keep) the stored data from container "from" to container "to"
594  m_mergeKeepFunctions.emplace_back( []( const VariableSet& from, VariableSet& to ) {
595  if ( !to.existsVariableType<T>() ) { to.addVariableType<T>(); }
596  for ( const auto& t : m_variables<T>[&from] ) {
597  m_variables<T>[&to].insert( t );
598  }
599  } );
600  // used to merge (replace) the stored data from container "from" to container "to"
601  m_mergeReplaceFunctions.emplace_back( []( const VariableSet& from, VariableSet& to ) {
602  if ( !to.existsVariableType<T>() ) { to.addVariableType<T>(); }
603  for ( const auto& t : m_variables<T>[&from] ) {
604  m_variables<T>[&to].insert_or_assign( t.first, t.second );
605  }
606  } );
607  // use to compute gauge on the stored data
608  m_sizeFunctions.emplace_back(
609  []( const VariableSet& c ) { return m_variables<T>[&c].size(); } );
610  m_visitFunctions.emplace_back(
611  []( const VariableSet& c, const DynamicVisitorBase& v )
612  -> std::pair<bool, std::function<void( DynamicVisitorBase&, std::any && )>> {
613  auto id = getTypeIndex<T>();
614  if ( v.accept( id ) ) {
615  auto coll = std::ref( m_variables<T>[&c] );
616  return { true, [coll]( DynamicVisitorBase& visitor, std::any&& userParam ) {
617  for ( auto&& t : coll.get() ) {
618  visitor( { std::ref( t ) }, std::forward<std::any>( userParam ) );
619  }
620  } };
621  }
622  else {
623  return { false, nullptr };
624  }
625  } );
626 
627  // remember the stored type and its rank
628  m_storedType.emplace_back( std::type_index( typeid( T ) ) );
629 }

637 template <typename T>
638 bool VariableSet::deleteAllVariables() {
639  auto iter = m_variables<T>.find( this );
640  if ( iter != m_variables<T>.end() ) {
641  auto tidx = std::type_index( typeid( T ) );
642  auto it = std::find( m_storedType.begin(), m_storedType.end(), tidx );
643  auto idx = it - m_storedType.begin();
644  m_clearFunctions.erase( m_clearFunctions.begin() + idx );
645  m_copyFunctions.erase( m_copyFunctions.begin() + idx );
646  m_moveFunctions.erase( m_moveFunctions.begin() + idx );
647  m_mergeKeepFunctions.erase( m_mergeKeepFunctions.begin() + idx );
648  m_mergeReplaceFunctions.erase( m_mergeReplaceFunctions.begin() + idx );
649  m_sizeFunctions.erase( m_sizeFunctions.begin() + idx );
650  m_visitFunctions.erase( m_visitFunctions.begin() + idx );
651  m_storedType.erase( it );
652  m_variables<T>.erase( iter );
653  return true;
654  }
655  return false;
656 }
