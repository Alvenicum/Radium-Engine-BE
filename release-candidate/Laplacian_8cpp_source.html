40LaplacianMatrix cotangentWeightLaplacian( const VectorArray<Vector3>& p,
41 const AlignedStdVector<Vector3ui>& T ) {
42 LaplacianMatrix L( p.size(), p.size() );
43 for ( const auto& t : T ) {
44 uint i = t( 0 );
45 uint j = t( 1 );
46 uint k = t( 2 );
47 Vector3 IJ = p[j] - p[i];
48 Vector3 JK = p[k] - p[j];
49 Vector3 KI = p[i] - p[k];
50 Scalar cotI = Math::cotan( IJ, ( -KI ).eval() );
51 Scalar cotJ = Math::cotan( JK, ( -IJ ).eval() );
52 Scalar cotK = Math::cotan( KI, ( -JK ).eval() );
53 L.coeffRef( i, j ) -= cotK;
54 L.coeffRef( j, i ) -= cotK;
55 L.coeffRef( j, k ) -= cotI;
56 L.coeffRef( k, j ) -= cotI;
57 L.coeffRef( k, i ) -= cotJ;
58 L.coeffRef( i, k ) -= cotJ;
59 L.coeffRef( i, i ) += cotJ + cotK;
60 L.coeffRef( j, j ) += cotI + cotK;
61 L.coeffRef( k, k ) += cotI + cotJ;
62 }
63 return ( 0.5 * L );
64}

79Vector3 cotangentWeightLaplacian( const Vector3& v, const VectorArray<Vector3>& p ) {
80 Vector3 L;
81 L.setZero();
82 uint N = p.size();
83 Utils::CircularIndex i;
84 i.setSize( N );
85 for ( uint j = 0; j < N; ++j ) {
86 i.setValue( j );
87 Scalar cot_a = Math::cotan( ( v - p[i - 1] ), ( p[i] - p[i - 1] ) );
88 Scalar cot_b = Math::cotan( ( v - p[i + 1] ), ( p[i] - p[i + 1] ) );
89 L += ( cot_a + cot_b ) * ( v - p[i] );
90 }
91 return ( 0.5 * L );
92}
