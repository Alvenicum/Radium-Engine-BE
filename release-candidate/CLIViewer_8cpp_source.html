162 void CLIViewer::setCamera( Ra::Core::Utils::Index camIdx ) {
163  auto cameraManager = static_cast<Ra::Engine::Scene::CameraManager*>(
164  Ra::Engine::RadiumEngine::getInstance()->getSystem( "DefaultCameraManager" ) );
165 
166  if ( camIdx.isInvalid() || cameraManager->count() <= size_t( camIdx ) ) {
167  m_camera = new Ra::Core::Asset::Camera( m_parameters.m_size[0], m_parameters.m_size[1] );
168  m_camera->setFOV( 60.0_ra * Ra::Core::Math::toRad );
169  m_camera->setZNear( 0.1_ra );
170  m_camera->setZFar( 100_ra );
171 
172  auto aabb = Ra::Engine::RadiumEngine::getInstance()->computeSceneAabb();
173  Scalar f = m_camera->getFOV();
174  Scalar a = m_camera->getAspect();
175  const Scalar r = ( aabb.max() - aabb.min() ).norm() / 2_ra;
176  const Scalar x = r / std::sin( f / 2_ra );
177  const Scalar y = r / std::sin( f * a / 2_ra );
178  Scalar d = std::max( std::max( x, y ), 0.001_ra );
179 
180  m_camera->setPosition(
181  Ra::Core::Vector3( aabb.center().x(), aabb.center().y(), aabb.center().z() + d ) );
182  m_camera->setDirection( Ra::Core::Vector3( 0_ra, 0_ra, -1_ra ) );
183  Scalar zfar =
184  std::max( d + ( aabb.max().z() - aabb.min().z() ) * 2_ra, m_camera->getZFar() );
185  m_camera->setZFar( zfar );
186  }
187  else {
188  cameraManager->activate( camIdx );
189  m_camera = cameraManager->getActiveCamera();
190  }
191 }
