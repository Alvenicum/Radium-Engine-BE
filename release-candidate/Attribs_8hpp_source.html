401 explicit ScopedLockState( AttribManager* a ) {
402 a->for_each_attrib( [this]( const auto& attr ) {
403 return ( v.push_back( std::make_pair( attr, attr->isLocked() ) ) );
404 } );
405 }
410 ~ScopedLockState() {
411 for ( auto& p : v ) {
412 if ( !p.second && p.first->isLocked() ) { p.first->unlock(); }
413 }
