164void AttribManager::copyAttributes( const AttribManager& m,
165 const AttribHandle<T>& attr,
166 Handle... attribs ) {
167 if ( m.isValid( attr ) ) {
168 // get attrib to copy
169 auto& a = m.getAttrib( attr );
170 // add new attrib
171 auto h = addAttrib<T>( a.getName() );
172 // copy attrib data
173 getAttrib( h ).setData( a.data() );
174 }
175 // deal with other attribs

278template <typename T>
279AttribHandle<T> AttribManager::addAttrib( const std::string& name ) {
280 // does the attrib already exist?
281 AttribHandle<T> h = findAttrib<T>( name );
282 if ( isValid( h ) ) return h;
283
284 // create the attrib
285 smart_pointer_type attrib = std::make_unique<Attrib<T>>( name );
286
287 // look for a free slot
288 auto it = std::find_if(
289 m_attribs.begin(), m_attribs.end(), []( const auto& attr ) { return !attr; } );
290 if ( it != m_attribs.end() ) {
291 it->swap( attrib );
292 h.m_idx = std::distance( m_attribs.begin(), it );
293 }
294 else {
295 m_attribs.push_back( std::move( attrib ) );
296 h.m_idx = m_attribs.size() - 1;
297 }
298 m_attribsIndex[name] = h.m_idx;
299 h.m_name = name;
300 ++m_numAttribs;
301
302 notify( name );
303 return h;
304}

306template <typename T>
307void AttribManager::removeAttrib( AttribHandle<T>& h ) {
308 auto c = m_attribsIndex.find( h.m_name );
309 if ( c != m_attribsIndex.end() ) {
310 Index idx = c->second;
311 m_attribs[idx].reset( nullptr );
312 m_attribsIndex.erase( c );
313 }
314 h.m_idx = Index::Invalid(); // invalidate whatever!
315 auto name = h.m_name;
316 h.m_name = ""; // invalidate whatever!
317 --m_numAttribs;
318 notify( name );
319}
