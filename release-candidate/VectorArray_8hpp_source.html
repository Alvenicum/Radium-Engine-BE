22 public:
24 using AlignedStdVector<V>::AlignedStdVector;

27 MatrixMap getMap() {

113// Notes :
114// Using a map for eigen integration was recommended by [1].
115// According to this document [2], it is necessary to use a special allocator to enforce
116// alignment...
117// But actually it is not necessary to do so in C++11 [3] (it is actually harmful
118// as it removes some vector features such as initializer lists).
119// Finally we use our own aligned allocator.
120
121// [1] https://forum.kde.org/viewtopic.php?f=74&t=126959
122// [2] http://eigen.tuxfamily.org/dox-devel/group__TopicStlContainers.html
123// [3] http://eigen.tuxfamily.org/bz/show_bug.cgi?id=829
