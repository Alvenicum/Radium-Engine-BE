25template <typename T>
26void ParameterSetEditor::addEnumParameterWidget( const std::string& key,
27 T initial,
28 Ra::Engine::Data::RenderParameters& params,
29 const json& metadata ) {
30 auto m = metadata[key];
31
32 std::string description = m.contains( "description" ) ? m["description"] : "";
33 std::string nm = m.contains( "name" ) ? m["name"] : key.c_str();
34 if ( auto ec = params.getEnumConverter( key ) ) {
35 auto items = ( *ec )->getEnumerators();
36 auto onEnumParameterStringChanged = [this, &params, &key]( const QString& value ) {
37 params.addParameter( key, value.toStdString() );
38 emit parameterModified( key );
39 };
40 addComboBox( nm,
41 onEnumParameterStringChanged,
42 params.getEnumString( key, initial ),
43 items,
44 description );
45 }
46 else {
47 LOG( Core::Utils::logWARNING )
48 << "ParameterSet don't have converter for enum " << key << " use index<>int instead.";
49
50 auto items = std::vector<std::string>();
51 items.reserve( m["values"].size() );
52 for ( const auto& value : m["values"] ) {
53 items.push_back( value );
54 }
55
56 auto onEnumParameterIntChanged = [this, &params, &key]( T value ) {
57 params.addParameter( key, value );
58 emit parameterModified( key );
59 };
60 addComboBox( nm, onEnumParameterIntChanged, initial, items, description );
61 }

64template <typename T>
65void ParameterSetEditor::addNumberParameterWidget( const std::string& key,
66 T initial,
67 Ra::Engine::Data::RenderParameters& params,
68 const json& metadata ) {
69
70 auto onNumberParameterChanged = [this, &params, &key]( T value ) {
71 params.addParameter( key, value );
72 emit parameterModified( key );
73 };
74 if ( metadata.contains( key ) ) {
75 auto m = metadata[key];
76 auto min = std::numeric_limits<T>::lowest();
77 auto max = std::numeric_limits<T>::max();
78
79 std::string description = m.contains( "description" ) ? m["description"] : "";
80 std::string nm = m.contains( "name" ) ? m["name"] : key.c_str();
81
82 if ( m.contains( "oneOf" ) ) {
83 // the variable has multiple valid bounds
84 std::vector<std::pair<T, T>> bounds;
85 bounds.reserve( m["oneOf"].size() );
86 for ( const auto& bound : m["oneOf"] ) {
87 auto mini = bound.contains( "minimum" ) ? T( bound["minimum"] ) : min;
88 auto maxi = bound.contains( "maximum" ) ? T( bound["maximum"] ) : max;
89 bounds.emplace_back( mini, maxi );
90 }
91 auto predicate = [bounds]( T value ) {
92 bool valid = false;
93 auto it = bounds.begin();
94 while ( !valid && it != bounds.end() ) {
95 valid = value >= ( *it ).first && value <= ( *it ).second;
96 ++it;
97 }
98 return valid;
99 };
100
101 addConstrainedNumberInput<T>(
102 nm, onNumberParameterChanged, initial, predicate, description );
103 }
104 else if ( m.contains( "minimum" ) && m.contains( "maximum" ) ) {
105 min = T( m["minimum"] );
106 max = T( m["maximum"] );
107 if constexpr ( std::is_floating_point_v<T> ) {
108 addPowerSliderInput( nm, onNumberParameterChanged, initial, min, max, description );
109 }
110 else {
111 addSliderInput( nm, onNumberParameterChanged, initial, min, max, description );
112 }
113 }
114 else {
115 min = m.contains( "minimum" ) ? T( m["minimum"] ) : min;
116 max = m.contains( "maximum" ) ? T( m["maximum"] ) : max;
117 addNumberInput<T>( nm, onNumberParameterChanged, initial, min, max, description );
118 }
119 }
120 else if ( m_showUnspecified ) {
121 addNumberInput<T>( key, onNumberParameterChanged, initial );
122 }
