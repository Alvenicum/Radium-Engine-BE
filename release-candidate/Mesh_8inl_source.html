207 // this observer is called each time an attrib is added or removed from m_mesh
208 auto attrib = m_mesh.getAttribBase( name );
209 // if attrib not nullptr, then it's an attrib add, so attach an observer to it
210
211 if ( attrib ) {
212 auto itr = m_handleToBuffer.find( name );
213 if ( itr == m_handleToBuffer.end() ) {
214 m_handleToBuffer[name] = m_dataDirty.size();

216 addToTranslationTable( name );

222 attrib->attach( AttribObserver( this, idx ) );
223 }
224 // else it's an attrib remove, do nothing, cleanup will be done in updateGL()
225 else {

277void CoreGeometryDisplayable<T>::setupCoreMeshObservers() {
278 int idx = 0;
279 m_dataDirty.resize( m_mesh.vertexAttribs().getNumAttribs() );
280 m_vbos.resize( m_mesh.vertexAttribs().getNumAttribs() );
281 // here capture ref to idx to propagate idx incrementation

283 auto name = b->getName();
284 m_handleToBuffer[name] = idx;
285 m_dataDirty[idx] = true;

319template <typename CoreGeometry>
320void CoreGeometryDisplayable<CoreGeometry>::updateGL() {
321 if ( m_isDirty ) {
322 // Check that our dirty bits are consistent.
323 ON_ASSERT( bool dirtyTest = false; for ( auto d
324 : m_dataDirty ) { dirtyTest = dirtyTest || d; } );
325 CORE_ASSERT( dirtyTest == m_isDirty, "Dirty flags inconsistency" );
326 CORE_ASSERT( !( m_mesh.vertices().empty() ), "No vertex." );
327
328 updateGL_specific_impl();
329#ifdef CORE_USE_DOUBLE
330 // need convserion
331 auto func = [this]( Ra::Core::Utils::AttribBase* b ) {
332 auto idx = m_handleToBuffer[b->getName()];
333
334 if ( m_dataDirty[idx] ) {
335 if ( !m_vbos[idx] ) { m_vbos[idx] = globjects::Buffer::create(); }
336
337 auto stride = b->getStride();
338 auto eltSize = b->getNumberOfComponents();
339 auto size = b->getSize();
340 auto data = std::make_unique<float[]>( size * eltSize );
341 const void* ptr = b->dataPtr();
342 const char* cptr = reinterpret_cast<const char*>( ptr );
343
344 for ( size_t i = 0; i < size; i++ ) {
345 auto tptr = reinterpret_cast<const Scalar*>( cptr + i * stride );
346 for ( size_t j = 0; j < eltSize; ++j ) {
347 data[i * eltSize + j] = tptr[j];
348 }
349 }
350
351 m_vbos[idx]->setData(
352 size * eltSize * sizeof( float ), data.get(), GL_DYNAMIC_DRAW );
353
354 m_dataDirty[idx] = false;
355 }
356 };
357
358#else
359 auto func = [this]( Ra::Core::Utils::AttribBase* b ) {
360 auto idx = m_handleToBuffer[b->getName()];
361
362 if ( m_dataDirty[idx] ) {
363 if ( !m_vbos[idx] ) { m_vbos[idx] = globjects::Buffer::create(); }
364 m_vbos[idx]->setData( b->getBufferSize(), b->dataPtr(), GL_DYNAMIC_DRAW );
365 m_dataDirty[idx] = false;
366 }
367 };
368#endif
369 m_mesh.vertexAttribs().for_each_attrib( func );
370
371 // cleanup removed attrib
372 for ( auto buffer : m_handleToBuffer ) {
373 // do not remove name from handleToBuffer to keep index ...
374 // we could also update handleToBuffer, m_vbos, m_dataDirty
375 if ( !m_mesh.hasAttrib( buffer.first ) && m_vbos[buffer.second] ) {
376 m_vbos[buffer.second].reset( nullptr );
377 m_dataDirty[buffer.second] = false;
378 }
379 }
380
381 GL_CHECK_ERROR;
382 m_isDirty = false;
383 }

599template <typename T>
600void GeneralMesh<T>::triangulate() {
601 m_triangleIndices.clear();
602 m_triangleIndices.reserve( this->m_mesh.getIndices().size() );
603 for ( const auto& face : this->m_mesh.getIndices() ) {
604 if ( face.size() == 3 ) { m_triangleIndices.push_back( face ); }
605 else {
607 int minus { int( face.size() ) - 1 };
608 int plus { 0 };
609 while ( plus + 1 < minus ) {
610 if ( ( plus - minus ) % 2 ) {
611 m_triangleIndices.emplace_back( face[plus], face[plus + 1], face[minus] );
612 ++plus;
613 }
614 else {
615 m_triangleIndices.emplace_back( face[minus], face[plus], face[minus - 1] );
616 --minus;
617 }
618 }
619 }
620 }
