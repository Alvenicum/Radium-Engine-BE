83 void SkeletonComponent::handleAnimationLoading(
84  const std::vector<Core::Asset::AnimationData*>& data ) {
85  CORE_ASSERT( ( m_skel.size() != 0 ), "A Skeleton should be loaded first." );
86  m_animations.clear();
87  m_animations.reserve( data.size() );
88 
89  auto pose = m_skel.getPose( SpaceType::LOCAL );
90  for ( uint n = 0; n < data.size(); ++n ) {
91  m_animations.emplace_back();
92  m_animations.back().reserve( m_skel.size() );
93  auto handleAnim = data[n]->getHandleData();
94  for ( uint i = 0; i < m_skel.size(); ++i ) {
95  auto it =
96  std::find_if( handleAnim.cbegin(), handleAnim.cend(), [this, i]( const auto& ha ) {
97  return m_skel.getLabel( i ) == ha.m_name;
98  } );
99  if ( it == handleAnim.cend() ) {
100  m_animations.back().push_back( KeyFramedValue( 0_ra, pose[i] ) );
101  }
102  else {
103  m_animations.back().push_back( it->m_anim );
104  }
105  }
106  }
107  if ( m_animations.size() == 0 ) {
108  m_animations.emplace_back();
109  for ( uint i = 0; i < m_skel.size(); ++i ) {
110  m_animations[0].push_back( KeyFramedValue( 0_ra, pose[i] ) );
111  }
112  }
113  m_animationID = 0;
114  m_animationTime = 0_ra;
115 }

149 void SkeletonComponent::update( Scalar t ) {
150  m_wasReset = Core::Math::areApproxEqual( t, 0_ra );
151  if ( m_wasReset ) {
152  m_animationTime = t;
153  m_skel.setPose( m_refPose, SpaceType::LOCAL );
154 
155  updateDisplay();
156  return;
157  }
158 
159  m_animationTime = m_speed * t;
160  Scalar lastTime = 0;
161  for ( auto boneAnim : m_animations[m_animationID] ) {
162  lastTime = std::max( lastTime, *boneAnim.getTimes().rbegin() );
163  }
164  if ( m_autoRepeat ) {
165  if ( !m_pingPong ) { m_animationTime = std::fmod( m_animationTime, lastTime ); }
166  else {
167  m_animationTime = std::fmod( m_animationTime, 2 * lastTime );
168  if ( m_animationTime > lastTime ) { m_animationTime = 2 * lastTime - m_animationTime; }
169  }
170  }
171  else if ( m_pingPong ) {
172  if ( m_animationTime > 2 * lastTime ) { m_animationTime = 0_ra; }
173  else if ( m_animationTime > lastTime ) {
174  m_animationTime = 2 * lastTime - m_animationTime;
175  }
176  }
177 
178  // get the current pose from the animation
179  Core::Animation::Pose pose = m_skel.getPose( SpaceType::LOCAL );
180  if ( !m_animations.empty() ) {
181 #pragma omp parallel for
182  for ( int i = 0; i < int( m_animations[m_animationID].size() ); ++i ) {
183  pose[uint( i )] = m_animations[m_animationID][uint( i )].at(
184  m_animationTime, Core::Animation::linearInterpolate<Core::Transform> );
185  }
186  }
187  else {
188  pose = m_refPose;
189  }
190  m_skel.setPose( pose, SpaceType::LOCAL );
191 
192  updateDisplay();
193 }

289 void SkeletonComponent::setupSkeletonDisplay() {
290  m_renderObjects.clear();
291  m_boneDrawables.clear();
292  if ( !s_boneMesh ) {
293  s_boneMesh = std::make_shared<Data::Mesh>( "Bone Mesh" );
294  s_boneMesh->loadGeometry( makeBoneShape() );
295  s_boneMaterial.reset( new Data::BlinnPhongMaterial( "Bone Material" ) );
296  s_boneMaterial->m_kd = Color::Grey( 0.4_ra );
297  s_boneMaterial->m_ks = Color::Black();
298  s_boneRenderTechnique.reset( new Rendering::RenderTechnique );
299  s_boneRenderTechnique->setConfiguration(
300  *Data::ShaderConfigurationFactory::getConfiguration( "BlinnPhong" ) );
301  s_boneRenderTechnique->setParametersProvider( s_boneMaterial );
302  }
303  for ( uint i = 0; i < m_skel.size(); ++i ) {
304  if ( !m_skel.m_graph.isLeaf( i ) && !m_skel.m_graph.isRoot( i ) &&
305  m_skel.getLabel( i ).find( "_$AssimpFbx$_" ) == std::string::npos ) {
306  std::string name = m_skel.getLabel( i ) + "_" + std::to_string( i );
307  auto ro = new Engine::Rendering::RenderObject(
308  name, this, Rendering::RenderObjectType::Geometry );
309  ro->setRenderTechnique( s_boneRenderTechnique );
310  ro->setMesh( s_boneMesh );
311  ro->setMaterial( s_boneMaterial );
312  ro->setXRay( false );
313  addRenderObject( ro );
314  m_boneMap[m_renderObjects.back()] = i;
315  m_boneDrawables.push_back( ro );
316  }
317  else {
318  LOG( logDEBUG ) << "Bone " << m_skel.getLabel( i ) << " not displayed.";
319  }
320  }
321  updateDisplay();
322 }

345 void SkeletonComponent::updateDisplay() {
346  for ( auto& bone : m_boneDrawables ) {
347  uint boneIdx = m_boneMap.at( bone->getIndex() );
348  Core::Vector3 start;
349  Core::Vector3 end;
350  m_skel.getBonePoints( boneIdx, start, end );
351 
352  Core::Transform scale = Core::Transform::Identity();
353  scale.scale( ( end - start ).norm() );
354 
355  Core::Quaternion rot =
356  Core::Quaternion::FromTwoVectors( Core::Vector3::UnitZ(), end - start );
357 
358  Core::Transform boneTransform = m_skel.getTransform( boneIdx, SpaceType::MODEL );
359  Core::Matrix3 rotation = rot.toRotationMatrix();
360  Core::Transform drawTransform;
361  drawTransform.linear() = rotation;
362  drawTransform.translation() = boneTransform.translation();
363 
364  bone->setLocalTransform( drawTransform * scale );
365  }
366 }
