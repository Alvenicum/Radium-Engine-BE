17// Add a texture as material parameter with texture parameter set by default for this material
18inline TextureParameters& BlinnPhongMaterial::addTexture( const TextureSemantic& semantic,
19 const std::string& texture ) {
20 CORE_ASSERT( !texture.empty(), "Invalid texture name" );
21
22 TextureParameters data;
23 data.name = texture;
24 data.wrapS = GL_REPEAT;
25 data.wrapT = GL_REPEAT;
26 if ( semantic != TextureSemantic::TEX_NORMAL ) { data.minFilter = GL_LINEAR_MIPMAP_LINEAR; }
27 return addTexture( semantic, data );
