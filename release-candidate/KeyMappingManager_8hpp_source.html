141 bool operator<( const MouseBinding& b ) const {
142 return ( m_buttons < b.m_buttons ) ||

144 ( ( m_buttons == b.m_buttons ) && ( m_modifiers == b.m_modifiers ) &&

146 ( ( m_buttons == b.m_buttons ) && ( m_modifiers == b.m_modifiers ) &&
147 ( m_key == b.m_key ) && ( m_wheel < b.m_wheel ) );
148 }
